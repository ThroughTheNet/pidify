    def pid
      return nil unless pid_exists?
      dpid = nil
      begin
        File.open(pid_file, File::RDONLY) { |file| dpid = file.gets.chomp if file.flock(File::LOCK_SH|File::LOCK_NB); file.flock(File::LOCK_UN) }
      rescue
        return nil
      end
      return dpid.to_i if dpid && dpid.to_i > 0
      nil
    end
